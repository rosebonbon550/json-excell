    <script>
        document.addEventListener("DOMContentLoaded", () => {
            const fileInput = document.getElementById('fileInput');
            const dropText = document.getElementById('dropText');
            const previewBox = document.getElementById('preview-box');
            const convertBtn = document.getElementById('convertBtn');
            const proModal = document.getElementById('proModal');
            const codeProInput = document.getElementById('codeProInput');
            const validerCodeBtn = document.getElementById('validerCodeBtn');
            const adsBox = document.getElementById('adsBox');
            
            let jsonData = null;
            let originalName = 'tableau_donnees';
            let isProUser = false;

            validerCodeBtn.addEventListener('click', () => {
                if (codeProInput.value.trim() === "PREMIUM2026") {
                    isProUser = true;
                    if(adsBox) adsBox.style.display = 'none'; 
                    alert("Félicitations ! Version Premium activée avec succès. 🎉");
                } else {
                    alert("Code invalide.");
                }
            });

            window.checkPro = function(e) {
                if (!isProUser) {
                    e.preventDefault();
                    document.getElementById('formatCsv').checked = true;
                    proModal.style.display = 'flex';
                }
            };

            window.closeModal = function() { proModal.style.display = 'none'; };

            fileInput.addEventListener('change', (e) => {
                const file = e.target.files[0];
                if (!file) return;

                originalName = file.name.substring(0, file.name.lastIndexOf('.'));
                const reader = new FileReader();
                
                reader.onload = function(event) {
                    try {
                        const content = event.target.result;
                        jsonData = JSON.parse(content);
                        
                        // Si c'est un objet unique, on le met dans un tableau
                        if (!Array.isArray(jsonData)) {
                            jsonData = [jsonData];
                        }

                        previewBox.textContent = content.substring(0, 500) + (content.length > 500 ? '...' : '');
                        previewBox.style.display = 'block';
                        convertBtn.style.display = 'block';
                        dropText.textContent = `Fichier chargé : ${file.name}`;
                    } catch (err) {
                        alert("Erreur : Ce fichier n'est pas un JSON valide. Vérifiez sa structure.");
                        fileInput.value = '';
                    }
                };
                reader.readAsText(file);
            });

            convertBtn.addEventListener('click', () => {
                if (!jsonData || jsonData.length === 0) return;

                const format = document.querySelector('input[name="format"]:checked').value;
                
                if (format === 'excel' && isProUser) {
                    // Exportation en VRAI Excel grâce à la bibliothèque XLSX
                    const worksheet = XLSX.utils.json_to_sheet(jsonData);
                    const workbook = XLSX.utils.book_new();
                    XLSX.utils.book_append_sheet(workbook, worksheet, "Données");
                    XLSX.writeFile(workbook, originalName + '.xlsx');
                } else {
                    // Exportation en CSV classique (Gratuit)
                    const keys = Object.keys(jsonData[0]);
                    const csvRows = [keys.join(',')]; // En-tête
                    
                    for (const row of jsonData) {
                        const values = keys.map(key => {
                            const val = row[key] === null || row[key] === undefined ? '' : row[key];
                            return '"' + String(val).replace(/"/g, '""') + '"';
                        });
                        csvRows.push(values.join(','));
                    }
                    
                    const csvContent = "data:text/csv;charset=utf-8,\uFEFF" + csvRows.join('\n');
                    const encodedUri = encodeURI(csvContent);
                    const downloadLink = document.createElement('a');
                    downloadLink.href = encodedUri;
                    downloadLink.download = originalName + '.csv';
                    document.body.appendChild(downloadLink);
                    downloadLink.click();
                    document.body.removeChild(downloadLink);
                }
            });
        });
    </script>
